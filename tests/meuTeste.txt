int fatorial(int n) {
    se (n <= 1) entao
        retorne 1;
    senao
        retorne n * fatorial(n - 1);
}

int buscaBinaria(int vet[], int tamanho, int chave) {
    int baixo, alto;
    baixo = 0;
    alto = tamanho - 1;

    enquanto (baixo <= alto) execute {
        int meio;
        meio = (baixo + alto) / 2;

        se (vet[meio] == chave) entao
            retorne meio;
        senao se (vet[meio] < chave) entao
            baixo = meio + 1;
        senao
            alto = meio - 1;
    }

    retorne -1; /* Não encontrado */
}

int vet[5];
int fatorialDe;
int chave;

programa {
    int resultado;
    fatorialDe = 5;
    escreva "Fatorial de ";
    escreva fatorialDe;
    escreva " é ";
    escreva fatorial(fatorialDe);
    novalinha;

    vet[0] = 10;
    vet[1] = 20;
    vet[2] = 30;
    vet[3] = 40;
    vet[4] = 50;
    leia chave;

    resultado = buscaBinaria(vet, 5, chave);
    se (resultado != -1) entao {
        escreva "Elemento encontrado na posição: ";
        escreva resultado;
    } senao {
        escreva "Elemento não encontrado";
    }
    novalinha;

    escreva "Contagem até 10: ";
    {
        int i;
        i = 1;
        enquanto (i <= 10) execute {
            escreva i;
            escreva " ";
            i = i + 1;
        }
        novalinha;
    }
}
